UI Layer
Fragment · ViewBinding · Activity
›
Fragment observes StateFlow from ViewModel — never calls UseCases directly. Use repeatOnLifecycle(STARTED) for safe collection; it fully cancels the coroutine on stop (unlike launchWhenStarted). Always null out ViewBinding in onDestroyView.
HomeFragment.kt 
class HomeFragment : Fragment(R.layout.fragment_home) {

    private val viewModel: HomeViewModel by viewModels()
    private var _binding: FragmentHomeBinding? = null
    private val binding get() = _binding!!

    override fun onViewCreated(view: View, savedInstanceState: Bundle?) {
        super.onViewCreated(view, savedInstanceState)
        _binding = FragmentHomeBinding.bind(view)

        observeState()

        // Trigger actions — no business logic here
        binding.btnRefresh.setOnClickListener { viewModel.onRefresh() }
    }

    private fun observeState() {
        // repeatOnLifecycle cancels on STOP, resumes on START
        // safer than launchWhenStarted which only suspends
        viewLifecycleOwner.lifecycleScope.launch {
            repeatOnLifecycle(Lifecycle.State.STARTED) {
                viewModel.uiState.collect { state ->
                    when (state) {
                        is UiState.Loading -> showLoading()
                        is UiState.Success -> renderUsers(state.data)
                        is UiState.Error   -> showError(state.message)
                    }
                }
            }
        }
    }

    override fun onDestroyView() {
        super.onDestroyView()
        _binding = null  // avoid memory leaks with ViewBinding
    }
}
ViewModel
StateFlow · viewModelScope · Hilt
›
ViewModel owns UI state, coordinates UseCases, handles lifecycle. Never import Android UI framework — keeps it testable. Use sealed classes for exhaustive state.
HomeViewModel.kt 
@HiltViewModel
class HomeViewModel @Inject constructor(
    private val getUsersUseCase: GetUsersUseCase
) : ViewModel() {

    private val _uiState = MutableStateFlow<UiState>(UiState.Loading)
    val uiState: StateFlow<UiState> = _uiState.asStateFlow()

    init { loadUsers() }

    fun onRefresh() = loadUsers()

    private fun loadUsers() {
        viewModelScope.launch {
            _uiState.emit(UiState.Loading)
            getUsersUseCase()
                .onSuccess { _uiState.emit(UiState.Success(it)) }
                .onFailure { _uiState.emit(UiState.Error(it.message)) }
        }
    }
}

sealed class UiState {
    data object Loading : UiState()
    data class  Success(val data: List<User>) : UiState()
    data class  Error(val message: String?) : UiState()
}

Repository
Interface (domain) · Impl (data) · Single source of truth
›
Interface lives in domain, implementation in data. Repo decides: emit from cache → fetch remote → update Room → re-emit. Classic offline-first with Flow.
UserRepositoryImpl.kt 
// domain/repository/UserRepository.kt
interface UserRepository {
    suspend fun getUsers(): Result<List<User>>
    fun observeUsers(): Flow<List<User>>
}

// data/repository/UserRepositoryImpl.kt
class UserRepositoryImpl @Inject constructor(
    private val remote: UserRemoteSource,
    private val local:  UserDao
) : UserRepository {

    override fun observeUsers(): Flow<List<User>> =
        local.observeAll()   // Room Flow — auto re-emits on write

    override suspend fun getUsers(): Result<List<User>> =
        runCatching {
            val fresh = remote.fetchUsers()
            local.upsertAll(fresh.map { it.toEntity() })
            fresh.map { it.toDomain() }
        }
}

@Dao
interface UserDao {

    fun observeAll(): Flow<List<UserEntity>>  // reactive!

Common Patterns
🌊 Offline-First Flow Kotlin
flow { emit(local.get()) emit(remote.fetch() .also { local.save(it) }) }
offline_first.kt 
fun getUsersOfflineFirst(): Flow<Resource<List<User>>> = flow {
    emit(Resource.Loading)
    // 1. emit cached first (instant)
    val cached = local.getUsers()
    if (cached.isNotEmpty()) emit(Resource.Success(cached))

    // 2. fetch fresh
    runCatching { remote.fetchUsers() }
        .onSuccess {
            local.upsertAll(it)
            emit(Resource.Success(it))
        }
        .onFailure {
            // already emitted cache — just flag error
            emit(Resource.Error(it, cached))
        }
}

// Think: class UserViewModel : ViewModel()
class UserBloc extends Bloc<UserEvent, UserState> {
  final UserRepository _repo;

  UserBloc(this._repo) : super(UserLoading()) {
    // on<E> ≈ viewModelScope.launch + when(event)
    on<LoadUsers>((event, emit) async {
      try {
        final users = await _repo.getUsers();
        emit(UserLoaded(users));   // ≈ _state.emit(Success)
      } catch (e) {
        emit(UserError(e.toString()));
      }
    });
  }
}

// OkHttp SSE → callbackFlow → Flow<String>
fun streamCompletion(prompt: String): Flow<String> = callbackFlow {
    val request = Request.Builder()
        .url("https://api.anthropic.com/v1/messages")
        .post(buildBody(prompt))
        .build()

    val listener = object : EventSourceListener() {
        override fun onEvent(es, id, type, data) {
            if (data != "[DONE]") {
                trySend(parseChunk(data))  // non-blocking
            }
        }
        override fun onClosed(es) = close()
        override fun onFailure(es, t, r) = close(t)
    }
    val es = EventSources.createFactory(okHttp)
        .newEventSource(request, listener)

    awaitClose { es.cancel() }
}

// ViewModel — accumulate chunks into StateFlow
fun chat(prompt: String) {
    viewModelScope.launch {
        val buffer = StringBuilder()
        repo.streamCompletion(prompt).collect { chunk ->
            buffer.append(chunk)
            _uiState.emit(UiState.Streaming(buffer.toString()))
        }
        _uiState.emit(UiState.Done(buffer.toString()))
    }
}

class HomeViewModelTest {
    @get:Rule val mainDispatcherRule = MainDispatcherRule()

    private val useCase = mockk<GetUsersUseCase>()
    private lateinit var vm: HomeViewModel

    @Before fun setup() { vm = HomeViewModel(useCase) }

    @Test
    fun `emits success when use case returns data`() = runTest {
        val users = listOf(User("1", "Ada"))
        coEvery { useCase() } returns Result.success(users)

        vm.uiState.test {          // turbine
            skipItems(1)           // skip Loading
            val state = awaitItem()
            assertIs<UiState.Success>(state)
            assertEquals(users, state.data)
        }
    }
}

class SentimentRepository @Inject constructor() {

    private val analyzer = EntityExtraction
        .getClient(EntityExtractorOptions.Builder(ENGLISH).build())

    suspend fun extractEntities(text: String) =
        suspendCancellableCoroutine<List<EntityAnnotation>> { cont ->
            analyzer.annotate(EntityExtractionParams.Builder(text).build())
                .addOnSuccessListener { cont.resume(it) }
                .addOnFailureListener  { cont.resumeWithException(it) }
        }

    // TFLite — bring your own model
    private val interpreter = Interpreter(
        loadModelFile(assets, "model.tflite")
    )
    fun runInference(input: FloatArray): FloatArray {
        val output = Array(1) { FloatArray(10) }
        interpreter.run(arrayOf(input), output)
        return output[0]
    }
}
